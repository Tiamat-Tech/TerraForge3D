55static char *stristr4(const char *str, const char *pattern)
56{
57 size_t i;
58
59 if (!*pattern)
60 {
61 return (char *)str;
62 }
63
64 for (; *str; str++)
65 {
66 if (toupper((unsigned char)*str) == toupper((unsigned char)*pattern))
67 {
68 for (i = 1;; i++)
69 {
70 if (!pattern[i])
71 {
72 return (char *)str;
73 }
74
75 if (toupper((unsigned char)str[i]) != toupper((unsigned char)pattern[i]))
76 {
77 break;
78 }
79 }
80 }
81 }
82
83 return NULL;
84}
